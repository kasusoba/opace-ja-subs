# One Pace ASR bridge dependencies.
# Setup:  python -m venv .venv && source .venv/bin/activate && pip install -r requirements.txt
# Also needs ffmpeg on PATH (macOS: brew install ffmpeg).

faster-whisper    # ASR: Japanese transcription with word timestamps
rapidfuzz         # fuzzy string scoring for line rescue/interpolation
pysubs2           # read/write .ass/.srt/.vtt subtitle files
fugashi           # MeCab wrapper: kanji/kana -> hiragana readings for matching
unidic-lite       # the MeCab dictionary fugashi reads from
